Stored Procedures for tbl_dea_cust
1. Insert a new dealercustomer

USE AnyStore;
GO

CREATE PROCEDURE spInsertDeaCust
    @type VARCHAR(50),
    @name VARCHAR(150),
    @email VARCHAR(150),
    @contact VARCHAR(15),
    @address TEXT,
    @added_date DATETIME,
    @added_by INT
AS
BEGIN
    INSERT INTO tbl_dea_cust (type, name, email, contact, address, added_date, added_by)
    VALUES (@type, @name, @email, @contact, @address, @added_date, @added_by);
END;
GO

2. Retrieve all dealers/customers

USE AnyStore;
GO

CREATE PROCEDURE spGetAllDeaCust
AS
BEGIN
    SELECT * FROM tbl_dea_cust;
END;
GO

3. Update a dealer/customer

USE AnyStore;
GO

CREATE PROCEDURE spUpdateDeaCust
    @id INT,
    @type VARCHAR(50),
    @name VARCHAR(150),
    @email VARCHAR(150),
    @contact VARCHAR(15),
    @address TEXT,
    @added_date DATETIME,
    @added_by INT
AS
BEGIN
    UPDATE tbl_dea_cust
    SET type = @type,
        name = @name,
        email = @email,
        contact = @contact,
        address = @address,
        added_date = @added_date,
        added_by = @added_by
    WHERE id = @id;
END;
GO

4. Delete a dealer/customer

USE AnyStore;
GO

CREATE PROCEDURE spDeleteDeaCust
    @id INT
AS
BEGIN
    DELETE FROM tbl_dea_cust
    WHERE id = @id;
END;
GO
