Stored Procedures for tbl_products
1. Insert a new product

USE AnyStore;
GO

CREATE PROCEDURE spInsertProduct
    @name VARCHAR(50),
    @category VARCHAR(50),
    @description TEXT,
    @rate DECIMAL(18, 2),
    @qty DECIMAL(18, 2),
    @added_date DATETIME,
    @added_by INT
AS
BEGIN
    INSERT INTO tbl_products (name, category, description, rate, qty, added_date, added_by)
    VALUES (@name, @category, @description, @rate, @qty, @added_date, @added_by);
END;
GO

2. Retrieve all products

USE AnyStore;
GO

CREATE PROCEDURE spGetAllProducts
AS
BEGIN
    SELECT * FROM tbl_products;
END;
GO

3. Update a product

USE AnyStore;
GO

CREATE PROCEDURE spUpdateProduct
    @id INT,
    @name VARCHAR(50),
    @category VARCHAR(50),
    @description TEXT,
    @rate DECIMAL(18, 2),
    @qty DECIMAL(18, 2),
    @added_date DATETIME,
    @added_by INT
AS
BEGIN
    UPDATE tbl_products
    SET name = @name,
        category = @category,
        description = @description,
        rate = @rate,
        qty = @qty,
        added_date = @added_date,
        added_by = @added_by
    WHERE id = @id;
END;
GO

4. Delete a product

USE AnyStore;
GO

CREATE PROCEDURE spDeleteProduct
    @id INT
AS
BEGIN
    DELETE FROM tbl_products
    WHERE id = @id;
END;
GO
