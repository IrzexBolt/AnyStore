Stored Procedures for tbl_categories
1. Insert a new category

USE AnyStore;
GO

CREATE PROCEDURE spInsertCategory
    @title VARCHAR(50),
    @description TEXT,
    @added_date DATETIME,
    @added_by INT
AS
BEGIN
    INSERT INTO tbl_categories (title, description, added_date, added_by)
    VALUES (@title, @description, @added_date, @added_by);
END;
GO


2. Retrieve all categories
USE AnyStore;
GO

CREATE PROCEDURE spGetAllCategories
AS
BEGIN
    SELECT * FROM tbl_categories;
END;
GO

3. Update a category

USE AnyStore;
GO

CREATE PROCEDURE spUpdateCategory
    @id INT,
    @title VARCHAR(50),
    @description TEXT,
    @added_date DATETIME,
    @added_by INT
AS
BEGIN
    UPDATE tbl_categories
    SET title = @title,
        description = @description,
        added_date = @added_date,
        added_by = @added_by
    WHERE id = @id;
END;
GO

4. Delete a category

USE AnyStore;
GO

CREATE PROCEDURE spDeleteCategory
    @id INT
AS
BEGIN
    DELETE FROM tbl_categories
    WHERE id = @id;
END;
GO
