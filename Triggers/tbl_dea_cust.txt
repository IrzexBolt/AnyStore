Triggers for tbl_dea_cust
1. After Insert Trigger

USE AnyStore;
GO

CREATE TRIGGER trgAfterInsertDeaCust
ON tbl_dea_cust
AFTER INSERT
AS
BEGIN
    DECLARE @DeaCustId INT, @Name VARCHAR(150);
    SELECT @DeaCustId = id, @Name = name FROM inserted;
    
    PRINT 'A new dealer/customer with ID ' + CAST(@DeaCustId AS VARCHAR) + ' and name ' + @Name + ' has been added.';
END;
GO
